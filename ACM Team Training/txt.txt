#include<cstdio>
#include <iostream>
using namespace std;
int f[] = {0,0,1,3};
int main(void) {
	int r, g, b;
	while (cin >> r >> g >> b) {
		long long ans = 0;
		int front = 0, behind = 0;
		if (r) {
			front++;
			r--;
		}
		if (g) {
			front++;
			g--;
		}
		if (b) {
			front++;
			b--;
		}
		if (r) {
			behind++;
			r--;
		}
		if (g) {
			behind++;
			g--;
		}
		if (b) {
			behind++;
			b--;
		}
		ans += f[front];
		ans += front * behind + f[behind];
		ans += (front + behind)*(r + b + g);
		cout << ans << endl;
	}
	
	return 0;
}
